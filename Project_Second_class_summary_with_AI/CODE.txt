        // Reveal Animations on Scroll
        function reveal() {
            var reveals = document.querySelectorAll(".reveal");
            for (var i = 0; i < reveals.length; i++) {
                var windowHeight = window.innerHeight;
                var elementTop = reveals[i].getBoundingClientRect().top;
                var elementVisible = 150;
                if (elementTop < windowHeight - elementVisible) {
                    reveals[i].classList.add("active");
                }
            }
        }

        window.addEventListener("scroll", reveal);

        // Initial reveal
        window.onload = () => {
            reveal();
        };

        // Simulated "Learning Start" interaction
        function startLearning() {
            const bar = document.getElementById('loadBar');
            const status = document.getElementById('statusText');
            
            bar.style.width = '100%';
            status.innerText = "Initializing Modules...";
            status.style.color = "#22d3ee";

            setTimeout(() => {
                // In a real app, this would navigate to classes.html
                // For this demo, we'll just show a success state
                status.innerText = "Redirecting to Classroom...";
                setTimeout(() => {
                    location.href = 'classes.html';
                }, 1000);
            }, 1500);
        }
